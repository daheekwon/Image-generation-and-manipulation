2. Laplacian pyramid

  • I construct 4 levels of Laplacian pyramid for every every single frame in the video sequence.
  • Each levels (i) of pyramid is the residual of adjacent levels of Gaussian pyramid (i) and (i+1) and represents corresponding spatial frequency information.
pyramid_level = 4;
 
face_L_pyramid = cell(size(YIQframe_face,4),pyramid_level+1);
for i = 1:size(YIQframe_face,4)
face_L_pyramid(i,:) = genpyramid(YIQframe_face(:,:,:,i),pyramid_level,"laplace");
end

3.1 A Design of band pass filter to the signal

  • To get specific temporal hidden variations, we should do temporal filtering.
  • For temporal filtering process, I design a "butterworth" band pass filter with parameters (filter order= 1024, First 3dB frequecy=0.83, Second 3dB frequency=1, sample rate= 30).
  • These parameters are from the EVM paper. (In the paper, they use the "ideal filter". To approximate ideal filter, I increase the filter order.)
h = fdesign.bandpass('N,F3dB1,F3dB2',1024,0.83,1,30);
Hd = design(h, 'butter');
fftHd = freqz(Hd,301, 'whole');
% visualize the filter I designed.
fvtool(Hd);

3.2 Convert the time series to the frequency domain and Apply a band pass filter

5. Image Reconstructioin

  • Based on the above plots, I choose last two levels of pyramid to amplify.
  • I amplify the signals 50 times (alpha = 50) and the rest of levels are attenuated (alpha = 0).
  • Then, I reconstruct the Laplacian pyramid.
face_4 = convert(face_4);
face_5 = convert(face_5);
g_filter = [1 2 1; 2 4 2; 1 2 1]/16;
recon = zeros(size(frame_face));
 
for i = 1:size(frame_face,4)
temp = face_5{i}*50 + face_L_pyramid{i,5};
temp = imresize(temp,2);
temp = imfilter(temp,g_filter,"replicate","same") + face_L_pyramid{i,4};
temp = imresize(temp,2);
temp = imfilter(temp,g_filter,"replicate","same") + face_L_pyramid{i,3};
temp = imresize(temp,2);
temp = imfilter(temp,g_filter,"replicate","same") + face_L_pyramid{i,2};
temp = imresize(temp,2);
temp = imfilter(temp,g_filter,"replicate","same") + face_L_pyramid{i,1};
recon(:,:,:,i) = temp;
end
recon_face = convertYIQ2RGB(recon);
v = VideoWriter('reconstructed_face.mp4','MPEG-4');
open(v);
writeVideo(v,recon_face);
close(v);

Functions

function [double] = convertframedouble(frame)
 
double = zeros(size(frame));
for i = 1:size(frame,4)
double(:,:,:,i) = im2double(frame(:,:,:,i));
end
end
 
 
function [YIQ] = convertframeYIQ(frame)
 
YIQ = zeros(size(frame));
for i = 1:size(frame,4)
YIQ(:,:,:,i) = rgb2ntsc(frame(:,:,:,i));
end
 
end
 
function [tmp_frame] = convertYIQ2RGB(YIQ)
tmp_frame = zeros(size(YIQ));
 
for i = 1:size(YIQ,4)
frame = ntsc2rgb(YIQ(:,:,:,i));
 
frame(frame > 1) = 1;
frame(frame < 0) = 0;
tmp_frame(:,:,:,i) = frame;
end
 
end
 
function [pyramid] = genpyramid(img,level,type)
% level <= 5
 
G_pyramid = cell(1,level+1);
G_pyramid{1} = img;
 
for i = 2:level+1
G_pyramid{i} = G_filter(G_pyramid{i-1});
end
 
L_pyramid = cell(1,level+1);
 
for i = 1:level
L_pyramid{i} = L_filter(G_pyramid{i},G_pyramid{i+1});
end
 
L_pyramid{level+1} = G_pyramid{level+1};
 
if strcmp(type,"gauss")
pyramid = G_pyramid{5};
elseif strcmp(type,"laplace")
pyramid = L_pyramid;
end
 
end
 
function [filter_img] = G_filter(img)
 
g_filter = [1 2 1; 2 4 2; 1 2 1]/16;
temp_img = imfilter(img,g_filter,"replicate","same");
filter_img = zeros(ceil(size(temp_img,1)/2),ceil(size(temp_img,2)/2),size(temp_img,3));
for j = 1:3
ttemp_img = temp_img(:,:,j);
filter_img(:,:,j) = ttemp_img(1:2:size(temp_img,1),1:2:size(temp_img,2));
end
 
end
 
function [Lap_img] = L_filter(ori,blur)
Lap_img = zeros(size(ori));
for j = 1:3
Lap_img(:,:,j) = ori(:,:,j) - imresize(blur(:,:,j),size(ori(:,:,j)));
end
end
 
function [Signal, Signal_filter] = Filtering(pyramid,level,i,j,k,filter)
 
Signal = zeros(size(pyramid,1),1);
for t = 1:size(pyramid,1)
Signal(t) = pyramid{t,level}(i,j,k);
end
 
Signal = fft(Signal);
Signal_filter = real(ifft(Signal.*filter));
 
%Signal = fftshift(fft(Signal));
%Signal_filter = real(ifft(ifftshift(Signal.*filter)));
% Signal_org = real(ifft(ifftshift(Signal)));
 
end
 
 
function [result] = convert(input)
result = cell(length(input{1,1,1}),1);
for t = 1:length(input{1,1,1})
result{t} = zeros(size(input));
for i = 1:size(input,1)
for j = 1:size(input,2)
for k = 1:size(input,3)
result{t}(i,j,k) = input{i,j,k}(t);
end
end
end
