HW2. Eulerian Video Magnification (Baby2 Video)

  • With eulerian video magnification, I amplify the color of baby2 video, which capture the color change of baby skin.
  • A detailed description of the experimental process is described in "HW2_face.mlx".

1. Initials and Color Transformations

1.1 Etract the frames of videos

1.2 Convert the frames into double-precision and YIQ color space

frame_baby2 = convertframedouble(frame_baby2);

2. Laplacian pyramid

  • I construct 6 levels of laplacian pyramid.
  • The level of the pyramid has been discovered through various experiments and is set to the optimal size.
pyramid_level = 6;
baby2_L_pyramid = cell(size(YIQframe_baby2,4),pyramid_level+1);
for i = 1:size(YIQframe_baby2,4)
baby2_L_pyramid(i,:) = genpyramid(YIQframe_baby2(:,:,:,i),pyramid_level,"laplace");
end

5. Image Reconstruction

  • I amplify the last 6th levels of filtered frame 50 times. (alpha = 100)
  • magnitude of amplification is empirically found.
baby2_6 = convert(baby2_6);
baby2_7 = convert(baby2_7);
g_filter = [1 2 1; 2 4 2; 1 2 1]/16;
recon = zeros(size(frame_baby2));
 
for i = 1:size(frame_baby2,4)
temp = baby2_7{i}*100 + baby2_L_pyramid{i,7};
temp = imresize(temp,[11,20]);
temp = imfilter(temp,g_filter,"replicate","same") + baby2_L_pyramid{i,6};
temp = imresize(temp,2);
temp = imfilter(temp,g_filter,"replicate","same") + baby2_L_pyramid{i,5};
temp = imresize(temp,2);

functions

function [double] = convertframedouble(frame)
 
double = zeros(size(frame));
for i = 1:size(frame,4)
double(:,:,:,i) = im2double(frame(:,:,:,i));
end
end
 
 
function [YIQ] = convertframeYIQ(frame)
 
YIQ = zeros(size(frame));
for i = 1:size(frame,4)
YIQ(:,:,:,i) = rgb2ntsc(frame(:,:,:,i));
end
 
end
 
function [tmp_frame] = convertYIQ2RGB(YIQ)
tmp_frame = zeros(size(YIQ));
 
for i = 1:size(YIQ,4)
frame = ntsc2rgb(YIQ(:,:,:,i));
 
frame(frame > 1) = 1;
frame(frame < 0) = 0;
tmp_frame(:,:,:,i) = frame;
end
 
end
 
function [pyramid] = genpyramid(img,level,type)
% level <= 5
 
G_pyramid = cell(1,level+1);
G_pyramid{1} = img;
 
for i = 2:level+1
G_pyramid{i} = G_filter(G_pyramid{i-1});
end
 
L_pyramid = cell(1,level+1);
 
for i = 1:level
L_pyramid{i} = L_filter(G_pyramid{i},G_pyramid{i+1});
end
 
L_pyramid{level+1} = G_pyramid{level+1};
 
if strcmp(type,"gauss")
pyramid = G_pyramid{5};
elseif strcmp(type,"laplace")
pyramid = L_pyramid;
end
 
end
 
function [filter_img] = G_filter(img)
 
g_filter = [1 2 1; 2 4 2; 1 2 1]/16;
temp_img = imfilter(img,g_filter,"replicate","same");
filter_img = zeros(ceil(size(temp_img,1)/2),ceil(size(temp_img,2)/2),size(temp_img,3));
for j = 1:3
ttemp_img = temp_img(:,:,j);
filter_img(:,:,j) = ttemp_img(1:2:size(temp_img,1),1:2:size(temp_img,2));
end
 
end
 
function [Lap_img] = L_filter(ori,blur)
Lap_img = zeros(size(ori));
for j = 1:3
Lap_img(:,:,j) = ori(:,:,j) - imresize(blur(:,:,j),size(ori(:,:,j)));
end
end
 
function [Signal, Signal_filter] = Filtering(pyramid,level,i,j,k,filter)
 
Signal = zeros(size(pyramid,1),1);
for t = 1:size(pyramid,1)
Signal(t) = pyramid{t,level}(i,j,k);
end
 
Signal = fft(Signal);
Signal_filter = real(ifft(Signal.*filter));
 
end
 
 
function [result] = convert(input)
result = cell(length(input{1,1,1}),1);
for t = 1:length(input{1,1,1})
result{t} = zeros(size(input));
for i = 1:size(input,1)
for j = 1:size(input,2)
for k = 1:size(input,3)
result{t}(i,j,k) = input{i,j,k}(t);
end
end
end
