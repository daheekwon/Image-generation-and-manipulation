HW2. Eulerian Video Magnification (Custom Video)

  • I take a video of my neck to capture the motion of pulse.
  • The video has 640*360*3 resolution with 384 frame.
  • I follow the EVM process (like HW description) while I empirically choose my own optimal parameters.

2. Laplacian pyramid

  • I use 4 levels of laplaciain pyramid. The pyramid consists of 4 residual images and one gaussian blurred image.
pyramid_level = 4;
custom_L_pyramid = cell(size(YIQframe_custom,4),pyramid_level+1);
for i = 1:size(YIQframe_custom,4)
custom_L_pyramid(i,:) = genpyramid(YIQframe_custom(:,:,:,i),pyramid_level,"laplace");
end

3.1 A Design of band pass filter to the signal

  • I use the bandpass butterworth filter with passing bound 0.7 and 1 Hz.
  • The filter order is 1024 (to approximate ideal filter) and the sampling rate is 30.
  • These parameters (especially 0.7 and 1) are based on the average pulse rate of an adult woman and empirically fine-tuned to get the best result.
h = fdesign.bandpass('N,F3dB1,F3dB2',1024,0.7,1,30);
Hd = design(h, 'butter');
fftHd = freqz(Hd,384,'whole');
% fdatool and fvtool써서 filter design and visualize도 해보기
fvtool(Hd);

5. Image Reconstruction

  • To reconstruct the video, I convert the format of filtered frames.
  • For the computational efficiecy, I only convert the frames that will be amplified (alpha !=0)
  • I set the alpha = 10
custom_4 = convert(custom_4);
custom_5 = convert(custom_5);
g_filter = [1 2 1; 2 4 2; 1 2 1]/16;
recon = zeros(size(frame_custom));
for i = 1:size(frame_custom,4)
temp = custom_5{i}*10 + custom_L_pyramid{i,5};
temp = imresize(temp,[80,45]);
temp = imfilter(temp,g_filter,"replicate","same") + custom_L_pyramid{i,4} + custom_4{i}*0;
temp = imresize(temp,2);
temp = imfilter(temp,g_filter,"replicate","same") + custom_L_pyramid{i,3};
temp = imresize(temp,2);
temp = imfilter(temp,g_filter,"replicate","same") + custom_L_pyramid{i,2};
temp = imresize(temp,2);
temp = imfilter(temp,g_filter,"replicate","same") + custom_L_pyramid{i,1};
recon(:,:,:,i) = temp;
end
recon_custom = convertYIQ2RGB(recon);
v = VideoWriter('reconstructed_custom.mp4','MPEG-4');
open(v)
writeVideo(v,recon_custom)
close(v)

functions

function [double] = convertframedouble(frame)
 
double = zeros(size(frame));
for i = 1:size(frame,4)
double(:,:,:,i) = im2double(frame(:,:,:,i));
end
end
 
 
function [YIQ] = convertframeYIQ(frame)
 
YIQ = zeros(size(frame));
for i = 1:size(frame,4)
YIQ(:,:,:,i) = rgb2ntsc(frame(:,:,:,i));
end
 
end
 
function [tmp_frame] = convertYIQ2RGB(YIQ)
tmp_frame = zeros(size(YIQ));
 
for i = 1:size(YIQ,4)
frame = ntsc2rgb(YIQ(:,:,:,i));
 
frame(frame > 1) = 1;
frame(frame < 0) = 0;
tmp_frame(:,:,:,i) = frame;
end
 
end
 
function [pyramid] = genpyramid(img,level,type)
% level <= 5
 
G_pyramid = cell(1,level+1);
G_pyramid{1} = img;
 
for i = 2:level+1
G_pyramid{i} = G_filter(G_pyramid{i-1});
end
 
L_pyramid = cell(1,level+1);
 
for i = 1:level
L_pyramid{i} = L_filter(G_pyramid{i},G_pyramid{i+1});
end
 
L_pyramid{level+1} = G_pyramid{level+1};
 
if strcmp(type,"gauss")
pyramid = G_pyramid{5};
elseif strcmp(type,"laplace")
pyramid = L_pyramid;
end
 
end
 
function [filter_img] = G_filter(img)
 
g_filter = [1 2 1; 2 4 2; 1 2 1]/16;
temp_img = imfilter(img,g_filter,"replicate","same");
filter_img = zeros(ceil(size(temp_img,1)/2),ceil(size(temp_img,2)/2),size(temp_img,3));
for j = 1:3
ttemp_img = temp_img(:,:,j);
filter_img(:,:,j) = ttemp_img(1:2:size(temp_img,1),1:2:size(temp_img,2));
end
 
end
 
function [Lap_img] = L_filter(ori,blur)
Lap_img = zeros(size(ori));
for j = 1:3
Lap_img(:,:,j) = ori(:,:,j) - imresize(blur(:,:,j),size(ori(:,:,j)));
end
end
 
function [Signal, Signal_filter] = Filtering(pyramid,level,i,j,k,filter)
 
Signal = zeros(size(pyramid,1),1);
for t = 1:size(pyramid,1)
Signal(t) = pyramid{t,level}(i,j,k);
end
 
Signal = fft(Signal);
Signal_filter = real(ifft(Signal.*filter));
 
end
 
 
function [result] = convert(input)
result = cell(length(input{1,1,1}),1);
for t = 1:length(input{1,1,1})
result{t} = zeros(size(input));
for i = 1:size(input,1)
for j = 1:size(input,2)
for k = 1:size(input,3)
result{t}(i,j,k) = input{i,j,k}(t);
end
end
end
end
